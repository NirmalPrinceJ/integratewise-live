d# IntegrateWise OS — 4-Day Execution Spec (96h)

> **Status**: Draft  
> **Version**: 1.0  
> **Last Updated**: 2026-01-28  
> **Architecture**: Truth + Context + AI Chats → Think → Governance → Act → Repeat

---

## Executive Summary

This document provides an implementation-grade specification for building a closed-loop operating system over 4 days (96 hours). The system ingests data from multiple sources (Truth, Context, AI Chats), normalizes it, produces actionable insights (Situations with A/B/C evidence), governs execution through policies, and executes actions through n8n/workers.

---

## Global Non-Negotiables (Applies to All 4 Days)

### Multi-tenant & Security Baseline

| Requirement | Implementation |
|-------------|----------------|
| Tenant isolation | Every record has `tenant_id` column |
| Request scoping | Every API call validates `tenant_id` from header + JWT claims |
| Idempotency | Every write operation uses `idempotency_key` to prevent duplicates |
| Governance gate | No execution without governance decision (unless auto-approve policy) |

### Evidence Model Baseline

All Think output must link to **A/B/C** evidence categories:

| Category | Source | Examples |
|----------|--------|----------|
| **A (Truth)** | Spine entities/events/metrics | Stripe payment events, HubSpot deal changes, CRM records |
| **B (Context)** | Artifacts/documents/chunks | Contracts, emails, screenshots, meeting transcripts |
| **C (AI Chats)** | IQ Hub sessions | Session summaries, extracted decisions, candidate actions |

### Shell Architecture Principle

> **"Shell does not compute"** — The BFF (Backend-for-Frontend) returns view-ready JSON only; all business logic remains in dedicated services (Think, Govern, Act, Knowledge).

interface RawEvent {
  tenant_id: string;

  idempotency_key: string;
  occurred_at: string;    // ISO timestamp
  payload: Record<string, any>;
  metadata?: Record<string, any>;
}

// Output: Normalized canonical record
interface NormalizedRecord {
  tenant_id: string;
  entity_type: 'client' | 'deal' | 'ticket' | 'account' | 'subscription' | 'transaction' | 'project' | 'task';
  source_system: string;
  source_id: string;
  canonical_id: string;   // stable UUID mapping
  payload_normalized: Record<string, any>;
  relationships?: Array<{
    target_type: string;
    target_id: string;
    role: string;         // parent, child, related
  }>;
  version: number;
  occurred_at: string;
}

export class Normalizer {
  private entityHandlers: Map<string, EntityHandler> = new Map();

  async normalize(event: RawEvent): Promise<NormalizedRecord | null> {
    // 1. Check idempotency - return existing if duplicate
    const existing = await this.checkIdempotency(event.tenant_id, event.idempotency_key);
    if (existing) return existing;

    // 2. Get appropriate handler
    const handler = this.entityHandlers.get(event.event_type);
    if (!handler) {
      await this.logError(event, `Unknown event_type: ${event.event_type}`);
      return null;
    }

    // 3. Validate with Zod schema
    const validationResult = handler.validate(event.payload);
    if (!validationResult.success) {
      await this.logError(event, `Validation failed: ${JSON.stringify(validationResult.errors)}`);
      return null;

    const normalized = await handler.transform(event);

    if (normalized.relationships) {

    }

    // 6. Upsert to Spine
    await this.writeToSpine(normalized);

// Think Output Contract
interface SituationDTO {
  id: string;
  scope: string;
  title: string;
  summary: string;
  severity: 'low' | 'medium' | 'high' | 'critical';
  confidence: number;
  entity?: { type: string; id: string; label: string };
  evidence_counts: { truth: number; context: number; ai_chats: number };
  proposed_actions: Array<{
    id: string;
    action_type: string;
    risk: 'low' | 'medium' | 'high';
    requires_approval: boolean;
    status: string;
  }>;
}

// Situation Templates (initial 5)
const SITUATION_TEMPLATES = [
  {
    trigger: { source_system: 'stripe', event_type: 'invoice.payment_failed' },
    scope: 'finance',
    title: (data: any) => `Payment failed for ${data.customer_id}`,
    severity: 'high',
    actions: [
      { action_type: 'email_customer', risk: 'low' },
      { action_type: 'create_ticket', risk: 'medium' },
    ],
  },
  {
    trigger: { source_system: 'hubspot', event_type: 'deal.stage_changed' },
    condition: (data: any) => ['closed_lost', 'negotiation'].includes(data.new_stage),
    scope: 'sales',
    title: (data: any) => `Deal slipped to ${data.new_stage}`,
    severity: 'medium',
    actions: [
      { action_type: 'notify_slack', risk: 'low' },
      { action_type: 'escalate', risk: 'high', requires_approval: true },
    ],
  },
  {
    trigger: { source_system: 'support', event_type: 'ticket_spike' },
    scope: 'operations',
    title: 'Support load rising',
    severity: 'medium',
    actions: [
      { action_type: 'create_ticket', risk: 'medium' },
    ],
  },
  {
    trigger: { source_system: 'finance', event_type: 'renewal_failed' },
    scope: 'cs',
    title: (data: any) => `Renewal risk: ${data.subscription_id}`,
    severity: 'critical',
    actions: [
      { action_type: 'email_customer', risk: 'low' },
      { action_type: 'schedule_meeting', risk: 'medium' },
      { action_type: 'escalate', risk: 'high', requires_approval: true },
    ],
  },
  {
    trigger: { source_system: 'operations', event_type: 'task_overload' },
    scope: 'operations',
    title: 'Capacity warning',
    severity: 'low',
    actions: [
      { action_type: 'create_ticket', risk: 'low' },
    ],
  },
];

export class ThinkEngine {
  async processEvent(normalizedEvent: NormalizedRecord): Promise<SituationDTO[]> {
    const situations: SituationDTO[] = [];

    // 1. Match event to templates
    const matchingTemplates = SITUATION_TEMPLATES.filter(
      (t) => t.trigger.source_system === normalizedEvent.source_system
    );

    for (const template of matchingTemplates) {
      // 2. Evaluate conditions
      if (template.condition && !template.condition(normalizedEvent.payload_normalized)) {
        continue;
      }

      // 3. Generate situation
      const situation = await this.createSituation(normalizedEvent, template);
      
      // 4. Gather A/B/C evidence
      const evidence = await this.gatherEvidence(normalizedEvent, situation);
      
      // 5. Create proposed actions
      const actions = await this.createActions(situation, template.actions);

      // 6. Persist
      await this.persistSituation(situation, evidence, actions);

      situations.push({
        ...situation,
        evidence_counts: {
          truth: evidence.filter((e) => e.plane === 'truth').length,
          context: evidence.filter((e) => e.plane === 'context').length,
          ai_chats: evidence.filter((e) => e.plane === 'ai_chats').length,
        },
        proposed_actions: actions.map((a) => ({
          id: a.id,
          action_type: a.action_type,
          risk: a.risk,
          requires_approval: a.requires_approval,
          status: a.status,
        })),
      });
    }

    return situations;
  }

  private async createSituation(
    event: NormalizedRecord,
    template: typeof SITUATION_TEMPLATES[0]
  ): Promise<Omit<SituationDTO, 'evidence_counts' | 'proposed_actions'>> {
    const title = typeof template.title === 'function'
      ? template.title(event.payload_normalized)
      : template.title;

    return {
      id: generateUUID(),
      scope: template.scope,
      title,
      summary: `Detected via ${event.source_system}: ${JSON.stringify(event.payload_normalized)}`,
      severity: template.severity,
      confidence: 0.85, // Initial confidence score
      entity: event.relationships?.[0] ? {
        type: event.relationships[0].target_type,
        id: event.relationships[0].target_id,
        label: event.payload_normalized.customer_id || event.payload_normalized.name,
      } : undefined,
    };
  }

  private async gatherEvidence(
    event: NormalizedRecord,
    situation: SituationDTO
  ): Promise<EvidenceRef[]> {
    const evidence: EvidenceRef[] = [];

    // A: Truth evidence (the event itself)
    evidence.push({
      plane: 'truth',
      ref_table: 'spine_events',
      ref_id: event.canonical_id,
      label: `${event.source_system} event: ${event.event_type}`,
    });

    // B: Context evidence (retrieve relevant documents)
    if (situation.entity) {
      const contextResults = await this.knowledgeService.retrieve({
        query: situation.title,
        entity: situation.entity,
        limit: 3,
      });
      evidence.push(
        ...contextResults.map((r) => ({
          plane: 'context' as const,
          ref_table: 'document_chunks',
          ref_id: r.chunk_id,
          label: `Document: ${r.filename}`,
          metadata: { snippet: r.snippet },
        }))
      );
    }

    // C: AI Chat evidence (IQ Hub sessions)
    const sessions = await this.iqHubService.search({
      query: situation.title,
      entity: situation.entity,
      limit: 2,
    });
    evidence.push(
      ...sessions.map((s) => ({
        plane: 'ai_chats' as const,
        ref_table: 'ai_sessions',
        ref_id: s.id,
        label: `AI Session: ${s.title}`,
      }))
    );

    return evidence;
  }

  private async createActions(
    situation: SituationDTO,
    templateActions: Array<{ action_type: string; risk: string; requires_approval?: boolean }>
  ): Promise<ProposedAction[]> {
    return templateActions.map((ta) => ({
      id: generateUUID(),
      situation_id: situation.id,
      action_type: ta.action_type,
      payload: this.buildActionPayload(ta.action_type, situation),
      risk: ta.risk as 'low' | 'medium' | 'high',
      requires_approval: ta.requires_approval ?? ta.risk !== 'low',
      status: 'proposed',
    }));
  }

  private buildActionPayload(actionType: string, situation: SituationDTO): Record<string, any> {
    const base = {
      situation_id: situation.id,
      entity: situation.entity,
      triggered_at: new Date().toISOString(),
    };

    switch (actionType) {
      case 'email_customer':
        return { ...base, template: 'payment_failed', to: situation.entity?.label };
      case 'create_ticket':
        return { ...base, priority: situation.severity === 'critical' ? 'urgent' : 'normal' };
      case 'notify_slack':
        return { ...base, channel: '#alerts', message: situation.title };
      case 'escalate':
        return { ...base, escalation_level: 'manager' };
      default:
        return base;
    }
  }
}

interface GovernanceRequest {
  proposed_action_id: string;
  action_type: string;
  risk: 'low' | 'medium' | 'high';
  entity_type?: string;
  scope: string;
  payload: Record<string, any>;
}

interface GovernanceDecision {
  decision: 'auto_approve' | 'require_approval' | 'deny';
  policy_id?: string;
  reason?: string;
}

export class PolicyEngine {
  async evaluate(request: GovernanceRequest): Promise<GovernanceDecision> {
    // 1. Fetch active policies ordered by priority
    const policies = await this.fetchPolicies(request.tenant_id, request.scope);

    // 2. Evaluate each policy in priority order
    for (const policy of policies) {
      if (!policy.enabled) continue;

      const match = this.matchesPolicy(request, policy.match);
      if (match) {
        return {
          decision: policy.decision,
          policy_id: policy.id,
          reason: `Matched policy: ${policy.name}`,
        };
      }
    }

    // 3. Default: require approval for non-low risk
    return {
      decision: request.risk === 'low' ? 'auto_approve' : 'require_approval',
      reason: 'Default policy: low risk auto-approve, others require approval',
    };
  }

  private matchesPolicy(request: GovernanceRequest, matchRule: Record<string, any>): boolean {
    if (matchRule.action_type && matchRule.action_type !== request.action_type) {
      return false;
    }
    if (matchRule.risk && matchRule.risk !== request.risk) {
      return false;
    }
    if (matchRule.entity_type && matchRule.entity_type !== request.entity_type) {
      return false;
    }
    // Additional conditions...
    return true;
  }
}

// API Routes
// POST /api/govern/decide
export async function decide(request: Request): Promise<Response> {
  const { proposed_action_id, decision, reason } = await request.json();
  const tenantId = getTenantId(request);

  // Validate decision
  if (!['approved', 'rejected'].includes(decision)) {
    return new Response(JSON.stringify({ error: 'Invalid decision' }), { status: 400 });
  }

  // Fetch current action state
  const action = await db.proposed_actions.find({ id: proposed_action_id, tenant_id: tenantId });
  if (!action) {
    return new Response(JSON.stringify({ error: 'Action not found' }), { status: 404 });
  }

  // Freeze evidence snapshot
  const evidenceSnapshot = await getEvidenceSnapshot(action.situation_id);

  // Create decision record
  await db.governance_decisions.create({
    tenant_id: tenantId,
    proposed_action_id,
    decision,
    decided_by: getUserId(request),
    reason,
    evidence_snapshot: evidenceSnapshot,
  });

  // Update action status
  await db.proposed_actions.update({
    id: proposed_action_id,
    tenant_id: tenantId,
    status: decision === 'approved' ? 'approved' : 'rejected',
  });

  return new Response(JSON.stringify({ success: true }));
}

export async function execute(request: Request): Promise<Response> {
  const { proposed_action_id } = await request.json();
  const tenantId = getTenantId(request);

  // 1. Verify action exists and is approved
  const action = await db.proposed_actions.find({
    id: proposed_action_id,
    tenant_id: tenantId,
  });

  if (!action) {
    return new Response(JSON.stringify({ error: 'Action not found' }), { status: 404 });
  }

  if (action.status !== 'approved') {
    return new Response(
      JSON.stringify({ error: 'Action not approved', required_status: 'approved', current_status: action.status }),
      { status: 403 }
    );
  }

  // 2. Create action run
  const run = await db.action_runs.create({
    tenant_id: tenantId,
    proposed_action_id,
    status: 'queued',
    executor: 'n8n', // or 'worker' based on action_type
  });

  // 3. Trigger executor (async)
  await this.triggerExecutor(run, action);

  return new Response(JSON.stringify({ run_id: run.id, status: run.status }));
}

private async triggerExecutor(run: ActionRun, action: ProposedAction): Promise<void> {
  await db.action_runs.update({
    id: run.id,
    status: 'running',
    started_at: new Date().toISOString(),
  });

  try {
    switch (action.executor) {
      case 'n8n':
        await this.triggerN8nWorkflow(run, action);
        break;
      case 'worker':
        await this.triggerWorker(run, action);
        break;
    }
  } catch (error) {
    await db.action_runs.update({
      id: run.id,
      status: 'failed',
      finished_at: new Date().toISOString(),
      error: { message: error.message, stack: error.stack },
    });
  }
}

// POST /api/act/runs/:id/retry
export async function retryRun(request: Request): Promise<Response> {
  const { id } = request.params;
  const tenantId = getTenantId(request);

  const run = await db.action_runs.find({ id, tenant_id: tenantId });
  if (!run) {
    return new Response(JSON.stringify({ error: 'Run not found' }), { status: 404 });
  }

  const action = await db.proposed_actions.find({ id: run.proposed_action_id, tenant_id: tenantId });

  // Create new attempt
  const newRun = await db.action_runs.create({
    tenant_id: tenantId,
    proposed_action_id: run.proposed_action_id,
    status: 'queued',
    attempt: run.attempt + 1,
    executor: run.executor,
  });

  await this.triggerExecutor(newRun, action);

  return new Response(JSON.stringify({ run_id: newRun.id, status: newRun.status }));
}
